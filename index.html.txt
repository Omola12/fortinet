    <script>
        function getComplaints() {
            return JSON.parse(localStorage.getItem('fortinet_complaints')) || [];
        }

        function saveComplaints(complaints) {
            localStorage.setItem('fortinet_complaints', JSON.stringify(complaints));
        }

        function generateId() {
            return 'FTNT-' + Date.now() + '-' + Math.random().toString(36).substr(2, 6).toUpperCase();
        }

        document.getElementById('complaintForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const name = document.getElementById('customerName').value;
            const email = document.getElementById('customerEmail').value;
            const subject = document.getElementById('subject').value;
            const category = document.getElementById('category').value;
            const message = document.getElementById('message').value;
            
            const complaintId = generateId();
            
            const complaint = {
                id: complaintId,
                name: name,
                email: email,
                subject: subject,
                category: category,
                message: message,
                status: 'Pending',
                reply: null,
                replyDate: null,
                date: new Date().toLocaleString(),
                company: 'Fortinet'
            };
            
            const complaints = getComplaints();
            complaints.push(complaint);
            saveComplaints(complaints);
            
            document.getElementById('message').innerHTML = `
                <div class="success">
                    ✅ Complaint submitted to Fortinet successfully!<br>
                    <strong>Your Complaint ID: ${complaintId}</strong><br>
                    Please save this ID to track your complaint status.<br>
                    <small>Fortinet support will respond within 24 hours.</small>
                </div>
            `;
            
            document.getElementById('complaintForm').reset();
            
            setTimeout(() => {
                document.getElementById('message').innerHTML = '';
            }, 10000);
        });
        
        function trackComplaint() {
            const trackingId = document.getElementById('trackingId').value.trim();
            if (!trackingId) {
                alert('Please enter your Complaint ID');
                return;
            }
            
            const complaints = getComplaints();
            const complaint = complaints.find(c => c.id === trackingId);
            
            const trackResult = document.getElementById('trackResult');
            
            if (!complaint) {
                trackResult.innerHTML = `<div class="error">❌ Complaint not found. Please check your Complaint ID and try again.</div>`;
                return;
            }
            
            let replyHtml = '';
            if (complaint.reply) {
                replyHtml = `
                    <div class="reply-box">
                        <strong>📝 Fortinet Support Response:</strong><br>
                        ${complaint.reply}<br>
                        <small>Response sent on: ${complaint.replyDate}</small>
                    </div>
                `;
            }
            
            trackResult.innerHTML = `
                <div class="complaint-status">
                    <strong>🏢 Company:</strong> Fortinet<br>
                    <strong>🎫 Complaint ID:</strong> ${complaint.id}<br>
                    <strong>📅 Date Submitted:</strong> ${complaint.date}<br>
                    <strong>📌 Subject:</strong> ${complaint.subject}<br>
                    <strong>📂 Category:</strong> ${complaint.category}<br>
                    <strong>📝 Your Message:</strong> ${complaint.message}<br>
                    <strong>⚡ Status:</strong> <span style="color: ${complaint.status === 'Resolved' ? 'green' : 'orange'}">${complaint.status}</span>
                    ${replyHtml}
                    <br><small>Thank you for choosing Fortinet.</small>
                </div>
            `;
        }
